\section*{Permutations with Repetition}

\begin{description}
    \item[Formula:] $$P_n(n_1, n_2, n_3, \dots, n_k) = \frac{n!}{n_1!n_2!n_3!\dots n_k!}$$
    \ 
    
    \item[Where:]
    \begin{itemize}
        \item Pn is the permutation with repetition of n elements of type $(n_1, n_2, \dots, n_k)$
        \item n = $n_1 + n_2 + n_3 + \dots + n_k$ is the total number of elements
        \item $n_1$ is the count of identical elements $a_1$
        \item $n_2$ is the count of identical elements $a_2$
        \item \dots
        \item $n_k$ is the count of identical elements $a_k$
    \end{itemize}
    \ 
\end{description}
